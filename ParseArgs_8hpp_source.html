121 std::vector<std::string> split(const std::string &values, const std::string &sep = ",",
122  bool trim_spaces = true, bool preserve_empty = true) {
123  auto remains = values;
124  std::vector<std::string> results;
125  std::string::size_type pos;
126 
127  while ((pos = remains.find(sep)) != std::string::npos) {
128  std::string value = remains.substr(0, pos);
129  if (trim_spaces) {
130  value = trim(value);
131  }
132  if (!value.empty() || preserve_empty) {
133  results.push_back(value);
134  }
135  remains = remains.substr(pos + 1);
136  }
137 
138  if (!remains.empty()) {
139  results.push_back(remains);
140  }
141  return results;
142 }

172  void parse() {
173  if (parsed_)
174  return;
175 
176  std::for_each(args_.begin(), args_.end(), [this](const std::string &s) {
177 
178  // TODO: would this be simpler, faster using std::regex?
179  VLOG(2) << "Parsing: " << s;
180  if (std::strspn(s.c_str(), "-") == 2) {
181  size_t pos = s.find('=', 2);
182 
183  std::string key, value;
184  if (pos != std::string::npos) {
185  key = s.substr(2, pos - 2);
186  if (key.length() == 0) {
187  LOG(ERROR) << "Illegal option value; no name for configuration: " << s;
188  return;
189  }
190  value = s.substr(pos + 1);
191  } else {
192  // Empty values are allowed for "flag-type" options (such as `--enable-log`).
193  // They can also "turn off" the flag, by pre-pending a `no-`: `--no-edit`)
194  key = s.substr(2, s.length());
195  if (key.find("no-") == 0) {
196  key = key.substr(3);
197  value = "off";
198  } else {
199  value = "on";
200  }
201  }
202  parsed_options_[key] = value;
203  VLOG(2) << key << " -> " << value;
204  } else {
205  VLOG(2) << "Positional(" << positional_args_.size() + 1 << "): " << s;
206  positional_args_.push_back(s);
207  }
208  });
209  parsed_ = true;
210  };
