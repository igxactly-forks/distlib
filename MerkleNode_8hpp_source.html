145 template<typename T, char *(hash_func)(const T &), size_t hash_len>
146 bool MerkleNode<T, hash_func, hash_len>::validate() const {
147 
148  // If either child is not valid, the entire subtree is invalid too.
149  if (left_ && !left_->validate()) {
150  return false;
151  }
152  if (right_ && !right_->validate()) {
153  return false;
154  }
155 
156  std::unique_ptr<const char> computedHash(hasChildren() ? computeHash() : hash_func(*value_));
157  return memcmp(hash_, computedHash.get(), len()) == 0;
158 }

161 template<typename T, char *(hash_func)(const T &), size_t hash_len>
162 bool operator==(const MerkleNode<T, hash_func, hash_len> &lhs,
163  const MerkleNode<T, hash_func, hash_len> &rhs) {
164  // First the trivial case of identity.
165  if (&lhs == &rhs) {
166  return true;
167  }
168 
169  // If either child is missing, or not equal, the nodes are different.
170  if (*lhs.left() != *rhs.left() || *lhs.right() != *rhs.right()) {
171  return false;
172  }
173 
174  // Finally, the hash values themselves.
175  return memcmp(lhs.hash(), rhs.hash(), hash_len) == 0;
176 }

196 // Recursive implementation of the build algorithm.
197 //
198 template<typename NodeType>
199 const NodeType *build_(NodeType *nodes[], size_t len) {
200 
201  if (len == 1) return new NodeType(nodes[0], nullptr);
202  if (len == 2) return new NodeType(nodes[0], nodes[1]);
203 
204  size_t half = len % 2 == 0 ? len / 2 : len / 2 + 1;
205  return new NodeType(build_(nodes, half), build_(nodes + half, len - half));
206 }

208 template<typename T, typename NodeType>
209 const NodeType *build(const std::list<T> &values) {
210 
211  NodeType *leaves[values.size()];
212  int i = 0;
213  for (auto value : values) {
214  leaves[i++] = new NodeType(value);
215  }
216 
217  return build_(leaves, values.size());
218 };

262  // This is the only function that we need to implement, in order to specialize the template to
263  // our concrete types: the hash is computed using the MD5 algorithm (as implemented by the
264  // OpenSSL library).
265  virtual const char *computeHash() const {
266  char buffer[2 * MD5_DIGEST_LENGTH];
267  unsigned char *computedHash;
268 
269  memcpy(buffer, left_->hash(), MD5_DIGEST_LENGTH);
270 
271  // If the right node is null, we just double the left node.
272  memcpy(buffer + MD5_DIGEST_LENGTH, right_ ? right_->hash() : left_->hash(), MD5_DIGEST_LENGTH);
273  size_t len = basic_hash(buffer, 2 * MD5_DIGEST_LENGTH, &computedHash);
274  assert(len == MD5_DIGEST_LENGTH);
275 
276  return (char *) computedHash;
277  }
