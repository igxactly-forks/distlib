62  virtual void onUpdate(Server *client) {
63  // Make sure client will be deleted, even if an exception is thrown.
64  std::unique_ptr<Server> ps(client);
65 
66  std::shared_ptr<ServerRecord> record = MakeRecord(*client);
67  alive_.insert(record);
68 
69  // If it was previously suspected of being unresponsive, this server is removed from the
70  // suspected list:
71  auto removed = suspected_.erase(record);
72  if (removed > 0) {
73  VLOG(2) << *client << " previously suspected; added back to healthy set";
74  } else {
75  VLOG(3) << "Received a ping from " << *client;

95  virtual void onReport(Server *sender, SwimReport *report) {
96  std::unique_ptr<Server> ps(sender);
97 
98  VLOG(2) << self() << " received: " << *report;
99  for (auto record : report->alive()) {
100  if (record.server() == self()) {
101  // yes, we know we are alive, thank you very much.
102  continue;
103  }
104 
105  std::shared_ptr<ServerRecord> pRecord(new ServerRecord(record));
106  pRecord->set_didgossip(false);
107 
108  auto was_suspected = suspected_.find(pRecord);
109  if (was_suspected != suspected_.end() &&
110  (*was_suspected)->timestamp() < record.timestamp()) {
111  suspected_.erase(pRecord);
112  alive_.insert(pRecord);
113  } else if (was_suspected == suspected_.end()) {
114  alive_.insert(pRecord);
115  }
116  }
117 
118  for (auto record : report->suspected()) {
119  if (record.server() == self()) {
120  // Reports of our death were greatly exaggerated.
121  VLOG(2) << *sender << " reported this server (" << self() << ") as 'suspected'; pinging";
122  SwimClient client(*sender, port());
123  client.Ping();
124  continue;
125  }
126 
127  std::shared_ptr<ServerRecord> pRecord(new ServerRecord(record));
128  if (suspected_.find(pRecord) == suspected_.end()) {
129 
130  pRecord->set_didgossip(false);
131  pRecord->set_timestamp(::utils::CurrentTime());
132 
133  auto was_alive = alive_.find(pRecord);
134  if (was_alive != alive_.end() && (*was_alive)->timestamp() < record.timestamp()) {
135  // This is genuine new news and we need to update our records.
136  suspected_.insert(pRecord);
137  alive_.erase(pRecord);
138  } else if (was_alive == alive_.end()) {
139  // If we are here, it means that we haven't seen this server before, and
140  // we should record it as "suspicious."
141  suspected_.insert(pRecord);
142  }
143  }
144  }
145 
146  VLOG(2) << "After merging, alive: " << alive_
147  << "; suspected: " << suspected_;
148  }
